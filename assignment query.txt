 SELECT gender, COUNT(*) as count FROM employees GROUP BY gender ORDER BY count DESC;
 
 SELECT date_format(hire_date,"%M"), COUNT(*) AS total_hires
FROM employees
GROUP BY hire_date
ORDER BY total_hires DESC;
